@startuml NearJob_Sequence_WhiteBox

' --- Style ---
skinparam backgroundColor #FFFFFF
skinparam sequenceMessageAlign center
skinparam ParticipantPadding 20

actor "Utilisateur\n(Candidat/Entreprise)" as User
participant "Interface\n(React/Vite)" as UI
participant "Routeur / Limiteur" as Router
participant "Middleware\nAuth/JWT" as Auth
participant "Contrôleur\n(Logic)" as Ctrl
participant "Modèles / ORM\n(Sequelize)" as ORM
database "Base de Données\n(MySQL)" as DB

== Flux Candidat : Postuler à une offre ==

User -> UI : Clique sur "Postuler"
UI -> Router : POST /api/jobs/:id/apply
activate Router

Router -> Auth : verifyToken(headers)
activate Auth
Auth -> Auth : Décoder JWT
Auth --> Router : Token Valide (userId, userType)
deactivate Auth

Router -> Ctrl : applyToJob(req, res)
activate Ctrl
Ctrl -> ORM : Job.findByPk(jobId)
ORM -> DB : SELECT * FROM Jobs WHERE...
DB --> ORM : Job Data
ORM --> Ctrl : Job Object

Ctrl -> ORM : Application.create({...})
ORM -> DB : INSERT INTO Applications...
DB --> ORM : OK (Success)
ORM --> Ctrl : Application Instance

Ctrl --> UI : 201 Created (Success Message)
deactivate Ctrl
deactivate Router

UI --> User : Affiche "Candidature envoyée !"

== Flux Entreprise : Publier une offre ==

User -> UI : Remplit le formulaire d'offre
UI -> Router : POST /api/jobs
activate Router

Router -> Auth : checkRole('company')
Auth --> Router : Authorized

Router -> Ctrl : createJob(req, res)
activate Ctrl
Ctrl -> ORM : Job.create(jobData)
ORM -> DB : INSERT INTO Jobs...
DB --> ORM : Job Created
ORM --> Ctrl : Job Object

Ctrl --> UI : 201 Created (Job Data)
deactivate Ctrl
deactivate Router

UI --> User : Redirige vers "Mes Offres"

@enduml
